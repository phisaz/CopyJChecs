package fr.free.jchecs.swg;

import java.awt.event.ActionEvent;

import javax.swing.Action;
import javax.swing.Icon;
import javax.swing.SwingUtilities;

import static fr.free.jchecs.swg.ResourceUtils.getImageIcon;

/**
 * Action de l'option "Charger une partie".
 *
 * @author David Cotton
 */
final class LoadGameAction extends AbstractUIAction {
    /**
     * Identifiant de la classe pour la sérialisation.
     */
    private static final long serialVersionUID = -1184816274839511750L;

    /**
     * Petite icône.
     */
    private static final Icon ICON16 = getImageIcon("loadPGN16.png");

    /**
     * Icône moyenne.
     */
    private static final Icon ICON22 = getImageIcon("loadPGN22.png");

    /**
     * Crée une nouvelle instance de l'action.
     *
     * @param pUI Référence du composant maitre de l'interface.
     */
    LoadGameAction(final UI pUI) {
        super("loadGame", pUI);

        putValue(Action.SMALL_ICON, ICON16);
        putValue(Action.LARGE_ICON_KEY, ICON22);
    }

    /**
     * Réaction face au déclenchement de l'action.
     *
     * @param pEvent Evènement signalant le déclenchement.
     */
    @Override
    public void actionPerformed(final ActionEvent pEvent) {
        SwingUtilities.invokeLater(new GameLoader(getUI()));
    }
}
